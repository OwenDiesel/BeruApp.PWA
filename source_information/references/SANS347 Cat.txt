            ws.Range("AE22") = "SEP"
        'CAT I (Seen as CAT II for SASOL)
        ElseIf (varDN >= 32 And varDN < 100 And varPressure * varDN >= 100000) Or (varDN >= 100 And varDN * varPressure >= 100000 And varDN * varPressure < 350000 And varPressure >= 50) Then
            ws.Range("AE22") = "CAT I"
        'CAT II
        ElseIf (varDN >= 100 And varDN < 250 And varPressure * varDN >= 350000) Or (varDN >= 250 And varDN * varPressure >= 350000 And varDN * varPressure < 500000 And varPressure >= 50) Then
            ws.Range("AE22") = "CAT II"
        'CAT III
        ElseIf (varDN >= 250 And varDN * varPressure >= 500000) Then
            ws.Range("AE22") = "CAT III"
        Else
            ws.Range("AE22") = "Undetermined"
        End If
    ElseIf ws.Range("AD21") = "Piping - Dangerous Liquids" Then
        'Not Regulated
        If varPressure < 50 Then
            ws.Range("AE22") = "Not Regulated"
        'SEP
        ElseIf (varPressure >= 50 And varDN < 25) Or (varDN >= 25 And varPressure >= 50 And varDN * varPressure < 200000) Then
            ws.Range("AE22") = "SEP"
        'CAT I (Seen as CAT II for SASOL)
        ElseIf (varDN * varPressure >= 200000 And varPressure < 1000 And varPressure >= 50) Then
            ws.Range("AE22") = "CAT I"
        'CAT II
        ElseIf (varDN >= 25 And varDN < 200 And varDN * varPressure >= 200000 And varPressure < 50000) Or (varDN >= 200 And varPressure >= 1000 And varPressure < 50000) Then
            ws.Range("AE22") = "CAT II"
        'CAT III
        ElseIf varPressure >= 50000 And varDN >= 25 Then
            ws.Range("AE22") = "CAT III"
        Else
            ws.Range("AE22") = "Undetermined"
        End If
    ElseIf ws.Range("AD21") = "Piping - Non Dangerous Liquids" Then
        'Not Regulated
        If varPressure < 50 Then
            ws.Range("AE22") = "Not Regulated"
        'SEP
        ElseIf (varPressure >= 50 And varDN < 200) Or (varDN >= 200 And varDN < 500 And varPressure >= 50 And varPressure * varDN < 500000) Or (varDN >= 500 And varPressure >= 50 And varPressure < 1000) Then
            ws.Range("AE22") = "SEP"
        'CAT I (Seen as CAT II for SASOL)
        ElseIf (varDN >= 200 And varDN < 500 And varDN * varPressure >= 500000 And varPressure < 50000) Or (varDN >= 500 And varPressure >= 1000 And varPressure < 50000) Then
            ws.Range("AE22") = "CAT I"
        'CAT II
        ElseIf varDN >= 200 And varPressure >= 50000 Then
            ws.Range("AE22") = "CAT II"
        Else
            ws.Range("AE22") = "Undetermined"
        End If
    End If
End Sub
